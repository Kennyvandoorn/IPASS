167  void flush()
168  {
169  auto k = bus.transaction(cs);
170  for (unsigned int i = 0; i < bufsize; ++i)
171  {
172  k.write((buffer[i] >> 8) & 0xFF);
173  k.write(buffer[i] & 0xFF);
174  }
175  }

178  void write_implementation(hwlib::xy pos, hwlib::color col)
179  {
180  buffer[pos.y * wsize.x + pos.x] =
181  ((static_cast<uint_fast16_t>(col.blue) & 0x00f8) << 8)
182  + ((static_cast<uint_fast16_t>(col.green) & 0x00fc) << 3)
183  + ((static_cast<uint_fast16_t>(col.red) & 0x00f8) >> 3);
184  }
